            import numpy as np
            import pandas as pd
        
            def remove_outliers(df, column, method='iqr', threshold=3):
            """
            Remove outliers de um DataFrame
            :param df: DataFrame Pandas
            :param column: Nome da coluna a ser analisada
            :param method: Método para remoção ('iqr' ou 'zscore')
            :param threshold: Limite para definir outliers no método Z-Score (padrão: 3 desvios padrão)
            :return: DataFrame sem os outliers
            """
            if method == 'zscore':
                from scipy import stats
                z_scores = np.abs(stats.zscore(df[column]))
                return df[z_scores < threshold]
            
            elif method == 'iqr':
                Q1 = df[column].quantile(0.25)
                Q3 = df[column].quantile(0.75)
                IQR = Q3 - Q1
                lower_bound = Q1 - 1.5 * IQR
                upper_bound = Q3 + 1.5 * IQR
                return df[(df[column] >= lower_bound) & (df[column] <= upper_bound)]
            
            else:
                raise ValueError("Método inválido. Use 'iqr' ou 'zscore'.")
        
          # Exemplo de uso
          data = {'valores': [10, 12, 15, 14, 100, 9, 11, 10, 300, 13, 14, 15]}
          df = pd.DataFrame(data)
          
          # Removendo outliers
          df_sem_outliers = remove_outliers(df, 'valores', method='iqr')
        
          print("Original:")
          print(df)
          print("\nSem outliers:")
          print(df_sem_outliers)
